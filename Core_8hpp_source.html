410  template <bool Align, typename TypeList>
411  constexpr auto sizeOfImpl() -> std::size_t
412  {
413  using namespace boost::mp11;
414 
415  std::size_t size = 0;
416  std::size_t maxAlign = 0;
417  mp_for_each<mp_transform<mp_identity, TypeList>>([&](auto e) constexpr
418  {
419  using T = typename decltype(e)::type;
420  if constexpr (Align)
421  {
422  roundUpToMultiple(size, alignof(T));
423  maxAlign = std::max(maxAlign, alignof(T));
424  }
425  size += sizeof(T);
426  });
427 
428  // final padding, so next struct can start right away
429  if constexpr (Align)
430  roundUpToMultiple(size, maxAlign);
431  return size;
432  }
