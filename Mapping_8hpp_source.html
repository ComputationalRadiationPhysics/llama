149  template <typename Tree, typename... Coords>
150  LLAMA_FN_HOST_ACC_INLINE auto getTreeBlobByte(const Tree& tree, const Tuple<Coords...>& treeCoord)
151  -> std::size_t
152  {
153  const auto firstArrayIndex = treeCoord.first.arrayIndex;
154  if constexpr (sizeof...(Coords) > 1)
155  {
156  constexpr auto firstChildIndex = decltype(treeCoord.first.childIndex)::value;
157  return getTreeBlobSize(tree.childs, firstArrayIndex)
158  + sumChildrenSmallerThan<firstChildIndex>(
159  tree,
160  std::make_index_sequence<tupleSize<typename Tree::ChildrenTuple>>{})
161  + getTreeBlobByte(get<firstChildIndex>(tree.childs), treeCoord.rest);
162  }
163  else
164  return sizeof(typename Tree::Type) * firstArrayIndex;
165  }
