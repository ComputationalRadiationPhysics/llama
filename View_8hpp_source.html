103  const VirtualDatum<RightView, RightBoundDatumDomain, RightOwnView>& right) -> LeftDatum&
104  {
105  using RightDatum = VirtualDatum<RightView, RightBoundDatumDomain, RightOwnView>;

109  using RightInnerCoord = decltype(rightCoord);
110  if constexpr (hasSameTags<
111  typename LeftDatum::AccessibleDatumDomain,
112  LeftInnerCoord,
113  typename RightDatum::AccessibleDatumDomain,
114  RightInnerCoord>)
115  {
116  Functor {}(left(leftCoord), right(rightCoord));
117  }
118  });
119  });
120  return left;
