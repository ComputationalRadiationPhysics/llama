56 template <typename VirtualDatum1, typename VirtualDatum2>
57 LLAMA_FN_HOST_ACC_INLINE void pPInteraction(VirtualDatum1 p1, VirtualDatum2 p2, FP ts)
58 {
59  // Creating tempory virtual datum object for distance on stack:
60  auto distance = p1(tag::Pos()) + p2(tag::Pos());
61  distance *= distance; // square for each element
62  const FP distSqr = EPS2 + distance(tag::X()) + distance(tag::Y()) + distance(tag::Z());
63  const FP distSixth = distSqr * distSqr * distSqr;
64  const FP invDistCube = 1.0f / std::sqrt(distSixth);
65  const FP s = p2(tag::Mass()) * invDistCube;
66  distance *= s * ts;
67  p1(tag::Vel()) += distance;
68 }

72 template <std::size_t ProblemSize, std::size_t Elems, std::size_t BlockSize>
73 struct UpdateKernel
74 {
75  template <typename Acc, typename View>
76  LLAMA_FN_HOST_ACC_INLINE void operator()(const Acc& acc, View particles, FP ts) const
77  {
78  [[maybe_unused]] auto sharedView = [&] {
79  if constexpr (USE_SHARED)
80  {
81  const auto sharedMapping = [&] {
82  if constexpr (USE_SHARED_TREE)
83  return llama::mapping::tree::Mapping{
84  typename View::ArrayDomain{BlockSize},
85  llama::Tuple{llama::mapping::tree::functor::LeafOnlyRT()},
86  typename View::DatumDomain{}};
87  else
88  return llama::mapping::SoA{typename View::ArrayDomain{BlockSize}, typename View::DatumDomain{}};
89  }();
90 
91  // if there is only 1 thread per block, avoid using shared
92  // memory
93  if constexpr (BlockSize / Elems == 1)
94  return llama::allocViewStack<View::ArrayDomain::rank, typename View::DatumDomain>();
95  else
96  {
97  constexpr auto sharedMemSize = llama::sizeOf<typename View::DatumDomain> * BlockSize;
98  auto& sharedMem = alpaka::block::shared::st::allocVar<std::byte[sharedMemSize], __COUNTER__>(acc);
99  return llama::View{sharedMapping, llama::Array{&sharedMem[0]}};
100  }
101  }
102  else
103  return int{}; // dummy
104  }();
105 
106  const auto ti = alpaka::idx::getIdx<alpaka::Grid, alpaka::Threads>(acc)[0u];
107  const auto tbi = alpaka::idx::getIdx<alpaka::Block, alpaka::Threads>(acc)[0];
108 
109  const auto start = ti * Elems;
110  const auto end = alpaka::math::min(acc, start + Elems, ProblemSize);
111  LLAMA_INDEPENDENT_DATA
112  for (std::size_t b = 0; b < (ProblemSize + BlockSize - 1u) / BlockSize; ++b)
113  {
114  const auto start2 = b * BlockSize;
115  const auto end2 = alpaka::math::min(acc, start2 + BlockSize, ProblemSize) - start2;
116  if constexpr (USE_SHARED)
117  {
118  LLAMA_INDEPENDENT_DATA
119  for (auto pos2 = decltype(end2)(0); pos2 + ti < end2; pos2 += BlockSize / Elems)
120  sharedView(pos2 + tbi) = particles(start2 + pos2 + tbi);
121  alpaka::block::sync::syncBlockThreads(acc);
122  }
123  LLAMA_INDEPENDENT_DATA
124  for (auto pos2 = decltype(end2)(0); pos2 < end2; ++pos2)
125  LLAMA_INDEPENDENT_DATA
126  for (auto i = start; i < end; ++i)
127  if constexpr (USE_SHARED)
128  pPInteraction(particles(i), sharedView(pos2), ts);
129  else
130  pPInteraction(particles(i), particles(start2 + pos2), ts);
131  if constexpr (USE_SHARED)
132  alpaka::block::sync::syncBlockThreads(acc);
133  }
134  }
135 };

139 template <std::size_t ProblemSize, std::size_t Elems>
140 struct MoveKernel
141 {
142  template <typename Acc, typename View>
143  LLAMA_FN_HOST_ACC_INLINE void operator()(const Acc& acc, View particles, FP ts) const
144  {
145  const auto ti = alpaka::idx::getIdx<alpaka::Grid, alpaka::Threads>(acc)[0];
146 
147  const auto start = ti * Elems;
148  const auto end = alpaka::math::min(acc, start + Elems, ProblemSize);
149 
150  LLAMA_INDEPENDENT_DATA
151  for (auto i = start; i < end; ++i)
152  particles(i)(tag::Pos()) += particles(i)(tag::Vel()) * ts;
153  }
