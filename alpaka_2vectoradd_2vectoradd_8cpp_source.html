40 template <std::size_t ProblemSize, std::size_t Elems>
41 struct AddKernel
42 {
43  template <typename Acc, typename View>
44  LLAMA_FN_HOST_ACC_INLINE void operator()(const Acc& acc, View a, View b) const
45  {
46  const auto ti = alpaka::idx::getIdx<alpaka::Grid, alpaka::Threads>(acc)[0];
47 
48  const auto start = ti * Elems;
49  const auto end = alpaka::math::min(acc, start + Elems, ProblemSize);
50 
51  LLAMA_INDEPENDENT_DATA
52  for (auto i = start; i < end; ++i)
53  {
54  a(i)(tag::X {}) += b(i)(tag::X {});
55  a(i)(tag::Y {}) -= b(i)(tag::Y {});
56  a(i)(tag::Z {}) *= b(i)(tag::Z {});
57  }
58  }
59 };
