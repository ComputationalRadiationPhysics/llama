74 template <std::size_t Elems, std::size_t KernelSize, std::size_t ElemsPerBlock>
75 struct BlurKernel
76 {
77  template <typename Acc, typename View>
78  LLAMA_FN_HOST_ACC_INLINE void operator()(const Acc& acc, View oldImage, View newImage) const
79  {
80  const auto ti = alpaka::idx::getIdx<alpaka::Grid, alpaka::Threads>(acc);
81 
82  [[maybe_unused]] auto sharedView = [&] {
83  if constexpr (SHARED)
84  {
85  // Using SoA for the shared memory
86  constexpr auto sharedChunkSize = ElemsPerBlock + 2 * KernelSize;
87  const auto sharedMapping = llama::mapping::tree::Mapping(
88  typename View::ArrayDomain{sharedChunkSize, sharedChunkSize},
89  llama::Tuple{llama::mapping::tree::functor::LeafOnlyRT()},
90  typename View::DatumDomain{});
91  constexpr auto sharedMemSize = llama::sizeOf<PixelOnAcc> * sharedChunkSize * sharedChunkSize;
92  auto& sharedMem = alpaka::block::shared::st::allocVar<std::byte[sharedMemSize], __COUNTER__>(acc);
93  return llama::View{sharedMapping, llama::Array{&sharedMem[0]}};
94  }
95  else
96  return int{}; // dummy
97  }();
98 
99  [[maybe_unused]] const auto bi = alpaka::idx::getIdx<alpaka::Grid, alpaka::Blocks>(acc);
100  if constexpr (SHARED)
101  {
102  constexpr auto threadsPerBlock = ElemsPerBlock / Elems;
103  const auto threadIdxInBlock = alpaka::idx::getIdx<alpaka::Block, alpaka::Threads>(acc);
104 
105  const std::size_t bStart[2]
106  = {bi[0] * ElemsPerBlock + threadIdxInBlock[0], bi[1] * ElemsPerBlock + threadIdxInBlock[1]};
107  const std::size_t bEnd[2] = {
108  alpaka::math::min(acc, bStart[0] + ElemsPerBlock + 2 * KernelSize, oldImage.mapping.arrayDomainSize[0]),
109  alpaka::math::min(acc, bStart[1] + ElemsPerBlock + 2 * KernelSize, oldImage.mapping.arrayDomainSize[1]),
110  };
111  LLAMA_INDEPENDENT_DATA
112  for (auto y = bStart[0]; y < bEnd[0]; y += threadsPerBlock)
113  LLAMA_INDEPENDENT_DATA
114  for (auto x = bStart[1]; x < bEnd[1]; x += threadsPerBlock)
115  sharedView(y - bi[0] * ElemsPerBlock, x - bi[1] * ElemsPerBlock) = oldImage(y, x);
116 
117  alpaka::block::sync::syncBlockThreads(acc);
118  }
119 
120  const std::size_t start[2] = {ti[0] * Elems, ti[1] * Elems};
121  const std::size_t end[2] = {
122  alpaka::math::min(acc, start[0] + Elems, oldImage.mapping.arrayDomainSize[0] - 2 * KernelSize),
123  alpaka::math::min(acc, start[1] + Elems, oldImage.mapping.arrayDomainSize[1] - 2 * KernelSize),
124  };
125 
126  LLAMA_INDEPENDENT_DATA
127  for (auto y = start[0]; y < end[0]; ++y)
128  LLAMA_INDEPENDENT_DATA
129  for (auto x = start[1]; x < end[1]; ++x)
130  {
131  auto sum = llama::allocVirtualDatumStack<PixelOnAcc>();
132  sum = 0;
133 
134  using ItType = long int;
135  const ItType iBStart = SHARED ? ItType(y) - ItType(bi[0] * ElemsPerBlock) : y;
136  const ItType iAStart = SHARED ? ItType(x) - ItType(bi[1] * ElemsPerBlock) : x;
137  const ItType i_b_end
138  = SHARED ? ItType(y + 2 * KernelSize + 1) - ItType(bi[0] * ElemsPerBlock) : y + 2 * KernelSize + 1;
139  const ItType i_a_end
140  = SHARED ? ItType(x + 2 * KernelSize + 1) - ItType(bi[1] * ElemsPerBlock) : x + 2 * KernelSize + 1;
141  LLAMA_INDEPENDENT_DATA
142  for (auto b = iBStart; b < i_b_end; ++b)
143  LLAMA_INDEPENDENT_DATA
144  for (auto a = iAStart; a < i_a_end; ++a)
145  {
146  if constexpr (SHARED)
147  sum += sharedView(std::size_t(b), std::size_t(a));
148  else
149  sum += oldImage(std::size_t(b), std::size_t(a));
150  }
151  sum /= FP((2 * KernelSize + 1) * (2 * KernelSize + 1));
152  newImage(y + KernelSize, x + KernelSize) = sum;
153  }
154  }
155 };

364  }
365

373  {
374  LLAMA_INDEPENDENT_DATA

376  chunkIt->virtualHost(y + KERNEL_SIZE, x + KERNEL_SIZE)
377  = hostChunkView[chunkNr](y + KERNEL_SIZE, x + KERNEL_SIZE);
378  }
379  chunkIt++;
380  }
381  chrono.printAndReset("Blur kernel");
382
