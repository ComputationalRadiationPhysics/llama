26 struct HeatEquationKernel
27 {
28  template <typename View>
29  void operator()(uint32_t idx, const View& uCurrBuf, View& uNextBuf, uint32_t extent, double dx, double dt) const
30  {
31  const auto r = dt / (dx * dx);
32  if (idx > 0 && idx < extent - 1u)
33  uNextBuf[idx] = uCurrBuf[idx] * (1.0 - 2.0 * r) + uCurrBuf[idx - 1] * r + uCurrBuf[idx + 1] * r;
34  }
35 };

37 // Exact solution to the test problem
38 // u_t(x, t) = u_xx(x, t), x in [0, 1], t in [0, T]
39 // u(0, t) = u(1, t) = 0
40 // u(x, 0) = sin(pi * x)
41 double exactSolution(double const x, double const t)
42 {
43  constexpr double pi = 3.14159265358979323846;
44  return std::exp(-pi * pi * t) * std::sin(pi * x);
45 }

47 auto main() -> int
48 {
49  // Parameters (a user is supposed to change numNodesX, numTimeSteps)
50  const auto numNodesX = 10000;
51  const auto numTimeSteps = 200000;
52  const auto tMax = 0.001;
53  // x in [0, 1], t in [0, tMax]
54  const auto dx = 1.0 / static_cast<double>(numNodesX - 1);
55  const auto dt = tMax / static_cast<double>(numTimeSteps - 1);
56 
57  const auto r = dt / (dx * dx);
58  if (r > 0.5)
59  {
60  std::cerr << "Stability condition check failed: dt/dx^2 = " << r << ", it is required to be <= 0.5\n";
61  return 1;
62  }
63 
64  const auto mapping = llama::mapping::AoS{llama::ArrayDomain{numNodesX}, DatumDomain{}};
65  auto uNext = llama::allocView(mapping);
66  auto uCurr = llama::allocView(mapping);
67 
68  // Apply initial conditions for the test problem
69  for (uint32_t i = 0; i < numNodesX; i++)
70  uCurr[i] = exactSolution(i * dx, 0.0);
71 
72  const auto start = std::chrono::high_resolution_clock::now();
73  HeatEquationKernel kernel;
74  for (int step = 0; step < numTimeSteps; step++)
75  {
76  for (auto i = 0; i < numNodesX; i++)
77  kernel(i, uCurr, uNext, numNodesX, dx, dt);
78 
79  // We assume the boundary conditions are constant and so these values
80  // do not need to be updated.
81  std::swap(uNext, uCurr);
82  }
83  const auto end = std::chrono::high_resolution_clock::now();
84  std::cout << "Runtime: " << std::chrono::duration<double>(end - start).count() << "s\n";
85

89  {

91  maxError = std::max(maxError, error);
92  }
93

96  if (resultCorrect)

98  std::cout << "Execution results correct!\n";
99  return 0;
100  }
101  else
102  {
103  std::cout << "Execution results incorrect: error = " << maxError << " (the grid resolution may be too low)\n";
104  return 2;
