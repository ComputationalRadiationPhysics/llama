100  std::string addLineBreaks(std::string raw)
101  {
102  boost::replace_all(raw, "<", "<\n");
103  boost::replace_all(raw, ", ", ",\n");
104  boost::replace_all(raw, " >", ">");
105  boost::replace_all(raw, ">", "\n>");
106  auto tokens = split(raw, '\n');
107  std::string result = "";
108  int indent = 0;
109  for (auto t : tokens)
110  {
111  if (t.back() == '>' || (t.length() > 1 && t[t.length() - 2] == '>'))
112  indent -= 4;
113  result += nSpaces(indent) + t + "\n";
114  if (t.back() == '<')
115  indent += 4;
116  }
117  return result;
